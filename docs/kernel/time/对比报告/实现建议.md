```rust
// kernel/src/arch/x86_64/driver/tsc_clocksource.rs

pub struct TSCClocksource {
    inner: SpinLock<TSCInner>,
}

struct TSCInner {
    data: ClocksourceData,
    self_ref: Weak<TSCClocksource>,
    tsc_khz: u64,
    unstable: bool,
}

impl Clocksource for TSCClocksource {
    fn read(&self) -> CycleNum {
        CycleNum::new(unsafe { x86::time::rdtsc() })
    }

    fn enable(&self) -> Result<i32, SystemError> {
        // 检查 invariant TSC
        if !Self::has_invariant_tsc() {
            return Err(SystemError::ENODEV);
        }
        Ok(0)
    }
}

impl TSCClocksource {
    /// 检测 Invariant TSC
    fn has_invariant_tsc() -> bool {
        let cpuid = x86::cpuid::CpuId::new();
        if let Some(extended_info) = cpuid.get_extended_function_info() {
            return extended_info.has_invariant_tsc();
        }
        false
    }

    /// 检测 TSC Deadline 支持
    fn has_tsc_deadline() -> bool {
        let cpuid = x86::cpuid::CpuId::new();
        if let Some(feature_info) = cpuid.get_feature_info() {
            return feature_info.has_tsc_deadline();
        }
        false
    }
}
```

```rust
pub struct HPETClocksource {

impl Clocksource for HPETClocksource {

// kernel/src/arch/x86_64/driver/hpet_clockevent.rs

pub struct HPETClockevent {
    inner: SpinLock<HPETClockeventInner>,
}

struct HPETClockeventInner {
    data: ClockeventData,
    timer_num: u8,  // HPET 定时器编号
    self_ref: Weak<HPETClockevent>,
}

impl Clockevent for HPETClockevent {
    fn set_next_event(&self, delta_ns: u64) -> Result<(), SystemError> {
        let hpet = hpet_instance();
        let freq = 1000000000 / hpet.period() as u64;
        let ticks = delta_ns * freq / 1000000000;

        // 设置比较值
        let (inner, timer_reg) = unsafe { self.timer_mut(self.timer_num)? };
        unsafe {
            timer_reg.write_comparator_value(ticks);
        }
        Ok(())
    }

    fn set_oneshot(&self) -> Result<(), SystemError> {
        let (inner, timer_reg) = unsafe { self.timer_mut(0)? };
        unsafe {
            timer_reg.write_config(0x0004);  // One-shot mode
        }
        Ok(())
    }
}
```

#### **3. Local APIC Timer Clockevent 实现**

```rust
// kernel/src/arch/x86_64/driver/apic_timer_clockevent.rs

pub struct LocalApicClockevent {
    inner: SpinLock<LocalApicClockeventInner>,
}

struct LocalApicClockeventInner {
    data: ClockeventData,
    cpu_id: ProcessorId,
    mode: LocalApicTimerMode,
    self_ref: Weak<LocalApicClockevent>,
}

impl Clockevent for LocalApicClockevent {
    fn set_next_event(&self, delta_ns: u64) -> Result<(), SystemError> {
        let cpu_khz = TSCManager::cpu_khz();
        let ticks = delta_ns * cpu_khz / 1000000 / 4;  // 除以分频器

        let mut timer = local_apic_timer_instance_mut(self.cpu_id);
        timer.set_initial_cnt(ticks);
        timer.start_current();
        Ok(())
    }

    fn set_oneshot(&self) -> Result<(), SystemError> {
        let mut timer = local_apic_timer_instance_mut(self.cpu_id);
        timer.setup_lvt(vector, false, LocalApicTimerMode::Oneshot);
        Ok(())
    }

    fn set_periodic(&self, interval_hz: u64) -> Result<(), SystemError> {
        let cpu_khz = TSCManager::cpu_khz();
        let interval_ms = 1000 / interval_hz;
        let ticks = cpu_khz * interval_ms / 4;

        let mut timer = local_apic_timer_instance_mut(self.cpu_id);
        timer.install_periodic_mode(ticks, 4);
        Ok(())
    }
}
```

### 4.3 驱动程序开发顺序

#### **阶段 1：基础架构（1-2周）**

1. **实现 Clockevent Trait**
   - 定义 `Clockevent` 和 `ClockeventData`
   - 实现 per-CPU 时钟事件设备管理器

2. **重构 APIC Timer**
   - 将现有的 `LocalApicTimer` 改造为 `Clockevent`
   - 实现 `set_next_event` 和 `set_oneshot`
   - 移除硬编码的中断处理

3. **测试 Clockevent 框架**
   - 验证 oneshot 模式
   - 验证周期模式切换

#### **阶段 2：TSC Clocksource（2-3周）**

1. **实现 TSC Clocksource**
   - 检测 invariant TSC
   - 实现 `read()` 方法（`rdtsc`）
   - 计算正确的 mult/shift

2. **TSC 稳定性检测**
   - 使用 HPET 作为 watchdog
   - 实现 per-CPU TSC 同步检测
   - 标记不稳定 TSC

3. **注册 TSC 为 Clocksource**
   - Rating=300（高于 HPET）
   - 启用后优先使用

4. **性能测试**
   - 对比 TSC vs ACPI PM 的读取延迟
   - 验证纳秒级精度

#### **阶段 3：HPET Clockevent（2周）**

1. **实现 HPET Clockevent**
   - 支持多个 HPET 定时器
   - 实现 per-CPU 分配
   - 配置中断路由

2. **HPET 作为 Clockevent**
   - 优先级低于 APIC Timer
   - 作为 APIC Timer 的备选

#### **阶段 4：高精度定时器（3-4周）**

1. **实现 Hrtimer 框架**
   - 红黑树实现（使用 `alloc::collections::BTreeMap` 或自实现）
   - 定时器到期中断处理
   - `hrtimer_start` / `hrtimer_cancel`

2. **集成到现有 Timer**
   - 保留低精度 Timer（用于 ms 级别）
   - 新增 `Hrtimer`（用于 ns/μs 级别）
   - 用户空间 API：`timer_create` with `CLOCK_MONOTONIC`

3. **Tickless 支持（NO_HZ）**
   - 当 CPU 空闲时停止 tick
   - 使用 hrtimer 编排下一个事件
   - 实现 `tick_nohz_handler`

#### **阶段 5：优化与测试（2-3周）**

1. **Timekeeping 优化**
   - 实现 seqcount 快速路径
   - 无锁时间读取
   - 减少锁竞争

2. **集成测试**
   - lmbench 时间测试
   - 网络延迟测试
   - 调度器延迟测试

3. **压力测试**
   - 多核并发
   - CPU 热插拔
   - 频率调节

#### **4. 性能优化集成点**

**关键优化：seqcount 快速路径**

```rust
// kernel/src/time/timekeeping.rs (新增)

use core::sync::atomic::{AtomicU32, Ordering};

pub struct TkReadBase {
    base: u64,
    cycle_last: CycleNum,
    mult: u32,
    shift: u32,
    clock: Arc<dyn Clocksource>,
}

pub struct TimekeeperFast {
    seq: AtomicU32,
    tkr: RwLock<TkReadBase>,
}

impl TimekeeperFast {
    /// 快速路径：无锁读取（仅 reader）
    #[inline(always)]
    pub fn ktime_get_real_fast_ns(&self) -> u64 {
        loop {
            let seq = self.seq.load(Ordering::Acquire);
            let tkr = self.tkr.read();

            let now = tkr.base + clocksource_cyc2ns(
                tkr.clock.read().div(tkr.cycle_last),
                tkr.mult,
                tkr.shift
            );

            if seq & 1 == 0 &&
               self.seq.load(Ordering::Acquire) == seq {
                return now;
            }
        }
    }
}
```
